3. Make API : 

    // Get Data
    app.get('/student',async (req,res) => {
        let db = getDb()
        let Student = db.collection('Student')
        let Sdata = await Student.find().toArray()

        res.send({
            data: Sdata
        })
    })
    // Create Data
    app.post('/student',async (req,res) => {
        let {name, age, marks, subject} = req.body
        let db = getDb();
        let Student = db.collection('Student')
        let Sdata = await Student.insertOne({
            name,
            age,
            subject,
            marks
        })
        res.send({
            massage:"Data Inserted Successfully",
            data: Sdata
        })
    })
    // Update Data
    app.post('/update',async (req,res) => {
        let {name, age, marks, subject} = req.body
        let db = getDb()
        let Student = db.collection('Student')
        let Sdata = await Student.updateOne({name}, {
            $set:{
                age,
                subject,
                marks
            }
        })

        res.send({
            massage: "Student updated Successfully",
            data: Sdata
        })
    })
    // Delete Data
    app.post('/delete',async (req,res) => {
        let {name} = req.body
        let db = getDb()
        let Student = db.collection('Student')
        let Sdata = await Student.deleteOne({name})

        res.send({
            massage:"Student Deleted Successfully",
            data: Sdata
        })
    })
